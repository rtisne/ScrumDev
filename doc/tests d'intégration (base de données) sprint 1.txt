FROM `project`
JOIN `member_relations`
ON `project`.`id` = `member_relations`.`project`
JOIN `user`
ON `member_relations`.`member` = `user`.`id`
WHERE `user`.`id` = 1
ORDER BY `project`.`title` ASC
